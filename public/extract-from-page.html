📊 从页面提取新闻数据

📋 操作步骤：

  1. 确保你已经打开了新闻页面（News页面）
  2. 在新闻页面按 F12 打开开发者工具
  3. 切换到 Console（控制台） 标签
  4. 复制下面的代码，粘贴到控制台，按回车运行
  5. 数据会自动下载为JSON文件

方法1：提取页面显示的文章数据（推荐）

// 提取页面上显示的6个文章模块
const articles = [];
const articleCards = document.querySelectorAll('.group.glass.rounded-2xl');

articleCards.forEach((card, index) => {
  const img = card.querySelector('img');
  const dateEl = card.querySelector('.text-gray-400');
  const titleEl = card.querySelector('.text-xl');
  
  if (img && titleEl) {
    articles.push({
      id: 'article-' + (index + 1),
      image: img.src,
      date: {
        'zh-CN': dateEl ? dateEl.textContent.trim() : '',
        'zh-TW': dateEl ? dateEl.textContent.trim() : '',
        'en': dateEl ? dateEl.textContent.trim() : ''
      },
      title: {
        'zh-CN': titleEl.textContent.trim(),
        'zh-TW': titleEl.textContent.trim(),
        'en': titleEl.textContent.trim()
      },
      description: {
        'zh-CN': '',
        'zh-TW': '',
        'en': ''
      },
      link: 'article-' + (index + 1)
    });
  }
});

console.log('提取到的文章数据：', articles);
console.log(JSON.stringify(articles, null, 2));

// 下载为文件
const blob = new Blob([JSON.stringify(articles, null, 2)], { type: 'application/json' });
const url = URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = 'featured-articles-extracted.json';
a.click();
URL.revokeObjectURL(url);

alert('✅ 数据已提取并下载！共 ' + articles.length + ' 篇文章');

方法2：强制保存当前显示的数据到localStorage

// 这个脚本会尝试从页面提取数据并保存到localStorage
// 注意：需要在新闻页面运行

// 提取文章数据
const extractedArticles = [];
const cards = document.querySelectorAll('.group.glass.rounded-2xl');

cards.forEach((card, index) => {
  const img = card.querySelector('img');
  const dateEl = card.querySelector('.text-gray-400');
  const titleEl = card.querySelector('.text-xl');
  const descEl = card.querySelectorAll('p')[1]; // 第二个p标签通常是描述
  
  if (img && titleEl) {
    const article = {
      id: 'featured-' + Date.now() + '-' + index,
      image: img.src,
      date: {
        'zh-CN': dateEl ? dateEl.textContent.trim() : '',
        'zh-TW': dateEl ? dateEl.textContent.trim() : '',
        'en': dateEl ? dateEl.textContent.trim() : ''
      },
      title: {
        'zh-CN': titleEl.textContent.trim(),
        'zh-TW': titleEl.textContent.trim(),
        'en': titleEl.textContent.trim()
      },
      description: {
        'zh-CN': descEl ? descEl.textContent.trim() : '',
        'zh-TW': descEl ? descEl.textContent.trim() : '',
        'en': descEl ? descEl.textContent.trim() : ''
      },
      link: 'article-' + (index + 1)
    };
    extractedArticles.push(article);
  }
});

// 保存到localStorage
if (extractedArticles.length > 0) {
  localStorage.setItem('featuredArticles', JSON.stringify(extractedArticles));
  console.log('✅ 已保存 ' + extractedArticles.length + ' 篇文章到 localStorage');
  console.log(JSON.stringify(extractedArticles, null, 2));
  
  // 下载备份
  const blob = new Blob([JSON.stringify(extractedArticles, null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'featured-articles-backup.json';
  a.click();
  URL.revokeObjectURL(url);
  
  alert('✅ 成功！\n- 已保存到 localStorage\n- 已下载备份文件\n- 共 ' + extractedArticles.length + ' 篇文章\n\n请刷新页面查看效果');
} else {
  alert('❌ 未找到文章数据，请确保在新闻页面运行此脚本');
}

⚠️ 重要提示：

请在 新闻页面（http://localhost:3000 点击News）运行上面的代码，不要在这个页面运行！
